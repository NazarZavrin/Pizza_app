  name VARCHAR(20) PRIMARY KEY NOT NULL,
  ingredients VARCHAR(100) NOT NULL,
  weight DECIMAL NOT NULL,
  price DECIMAL NOT NULL,
  is_vegetarian BOOLEAN DEFAULT false
)
